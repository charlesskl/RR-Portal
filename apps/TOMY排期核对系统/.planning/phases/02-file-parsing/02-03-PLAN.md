3. Replace the stub handler with:
   ```typescript
   router.post('/process', upload.fields([...]), async (req, res) => {
     const files = req.files as { [fieldname: string]: Express.Multer.File[] }
     const poFiles = files?.pos || []
     const scheduleFiles = files?.schedule || []

     // Process each PO PDF
     const results: FileResult[] = []
     for (const file of poFiles) {
       try {
         const data = await extractPO(file.buffer, file.originalname)
         results.push({ filename: file.originalname, status: 'done', data })
       } catch (err) {
         results.push({
           filename: file.originalname,
           status: 'error',
           data: null,
           error: err instanceof Error ? err.message : 'Unknown error'
         })
       }
     }

     // Process schedule Excel
     let schedule: ProcessResponse['schedule'] = null
     if (scheduleFiles.length > 0) {
       const file = scheduleFiles[0]
       try {
         const rows = await parseScheduleExcel(file.buffer)
         schedule = { filename: file.originalname, status: 'done', rowCount: rows.length }
       } catch (err) {
         schedule = {
           filename: file.originalname,
           status: 'error',
           rowCount: 0,
           error: err instanceof Error ? err.message : 'Unknown error'
         }
       }
     }

     const response: ProcessResponse = { files: results, schedule }
     res.json(response)
   })
   ```

4. Make the handler async (Express 5 supports async error handling).
