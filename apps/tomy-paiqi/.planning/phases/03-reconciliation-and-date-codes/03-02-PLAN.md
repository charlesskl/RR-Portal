```typescript
const COUNTRY_MAP: Record<string, string> = {
```

3. **buildScheduleIndex(rows):** Map<string, ScheduleRow[]> keyed by `normalize(tomyPO) + ":" + normalize(货号)`

4. **compareFields(poData, poItem, scheduleRow):** Returns FieldMismatch[] by comparing:
   - 国家: normalize + COUNTRY_MAP lookup on PO side, normalize on schedule side
   - 第三客户名称: normalize text comparison
   - 客跟单: normalize text comparison
   - TOMY PO: normalize text comparison
   - CUSTOMER PO: normalize text comparison
   - 货号: normalize text comparison
   - 数量: numeric equality (skip if PO value null)
   - 外箱: numeric equality (skip if PO value null)
   - 总箱数: ALWAYS SKIP (not in PDF, per COMP-10)
   - PO走货期: parse PO string with date-fns `parse(str, 'd MMM yyyy', new Date())`, compare year/month/day against schedule Date
   - 箱唛資料: generate expected value using rules (customerName→标准唛/待定, +TTT, +欧盟联盟), then compare with normalize against schedule value

5. **generate箱唛資料(customerName, qcInstructions):** Implements the business rules.

6. **reconcile(poDataList: POData[], scheduleRows: ScheduleRow[]):** Returns ReconciliationResult.
   - Build schedule index
   - For each POData, check for duplicate 货号 within its items → flag ambiguous
   - For each non-ambiguous POItem, lookup in schedule index
   - If found: compareFields → create RowMatchResult with status 'matched', **and populate dateCode: `generateDateCode(poItem.PO走货期, poItem.factoryCode)`**
   - If not found: add to unmatchedPOItems
   - Track which schedule rows were matched (for status column in Plan 03)

CRITICAL: When building each matched RowMatchResult, the dateCode field MUST be populated:
```typescript
const matchResult: RowMatchResult = {
```
